By pingtingpeng

发表于 2016-06-25

####1.有如下代码，解释Person、prototype、__proto__、p、constructor之间的关联
       

function Person(name){
this.name = name;
}
Person.prototype.sayName = function(){
console.log('My name is :' + this.name);
}
var p = new Person("若愚")
p.sayName();

       Person:一个函数，每个函数都有prototype属性，
       prototype属性指向函数的原型，
       这个原型里边有constructor,constructor指向Person，
       可以为原型对象添加其他属性和方法，所以原型里边还有新添加的属性和方法，
       p是person的实例，每个实例都有proto属性，这个属性指向创造它的函数的原型。

####2.上例中，对对象 p可以这样调用 p.toString()。toString是哪里来的? 画出原型图?并解释什么是原型链。
       首先每个类型都会有一个内部属性prototype，称为原型，原型的值要么是一个对象要么是NULL，如果它的值是一个对象，则这个对象也一定有自己的原型，这样就形成了一条线性的链，称为原型链。原型链的图示：

       代码如下：
String.prototype.getMostOften=function(){
this.arr=this.split(‘’);
this.count=[];//存储次数
this.arr2=[];//存储字符
var maxnum=1;
for (var i = 0; i < this.arr.length; i++) {
var nextnum=1;
for (var j=i+1; j < this.arr.length; j++) {
if(this.arr[i]==this.arr[j]){
nextnum++;
this.arr.splice(j,1);//删除相同项
j–;
}
}
this.count.push(nextnum);
if(nextnum>maxnum){
maxnum=nextnum;
this.arr2.push(this.arr[i]);
}
}
return this.arr2[this.arr2.length-1];//次数最多的字符为最后一个
}

####4.instanceOf有什么作用？内部逻辑是如何实现的？
       instanceOf用来判断一个对象是否是一个类型的实例
       如果一个对象__proto__等于另一个类型的prototype，那么这个对象就为此类型的一个实例
function isInstanceOf(obj,Func){
var proto=obj.proto;
do{
if(proto===Func.prototype){
return true;
}
if(!proto){//proto最后为NULL
return false;
}
}while(proto=proto.proto)
return false;
}

4.Object.create 有什么作用？兼容性如何？如何使用？

       Object.create(proto, [ propertiesObject ]),是ES5中提出的一种新的对象创建方式,
     兼容性：只兼容IE9+，Chrome5+，Firefox 4.0+，Opear 11.60+,Safari 5+
     第一个参数是要继承的原型或null,第二个参数可选,是对象的属性描述符，里面可以添加一些自定义的属性和方法。
       propertiesObject:默认都为false
       数据属性：
       writable:是否可任意写
       configurable：是否能够删除，是否能够被修改
       enumerable：是否能用 for in 枚举
       value：值
      访问属性：
       get(): 访问
       set(): 设置

var obj = {
    a:function(){
        console.log(100)
    },
    b:function(){
        console.log(200)
    },
    c:function(){
        console.log(300)
    }

};

 newObj = {};

var newObj = Object.create(obj,{
    t1:{
        value:'yupeng',
        writable:true
    },
    bar: {
        configurable: false,
        get: function() { return bar; },
        set: function(value) { bar=value }
    }

})

console.log(newObj.a());//100
console.log(newObj.t1);//yupeng
newObj.t1='yupeng1'
console.log(newObj.t1);//yupeng1  如果writable:false则为yupeng
newObj.bar=201;
console.log(newObj.bar)//通过set设值201

       判断实例化后的对象的属性是属于父类的还是自己的，如果是属于自己的，返回true,属于父类的，返回false

Person.prototype.getName = function(){

Male.prototype= new Person();

p1.lc='hz';

6.实现Object.create的 polyfill

//不支持第二个参数
if (typeof Object.create != 'function') {//首先判断是否兼容
Object.create = (function() {
var Temp = function() {};
return function (prototype) {
  if (arguments.length > 1) {
    throw Error('Second argument not supported');
  }
  if(prototype !== Object(prototype)) {
    throw TypeError('Argument must be an object');
  }
  if (prototype === null) {
    throw Error('null [[Prototype]] not supported');
  }
  Temp.prototype = prototype;
  var result = new Temp();
  Temp.prototype = null;
  return result;
};
 })();

function inherit(superType,subType){
var _prototype=Object.create(superType.prototype);//拷贝一份新的prototype
_prototype.constructor=subType;//将constructor指向子类
subType.prototype=_prototype;//将子类的prototype指向新的prototype

}
